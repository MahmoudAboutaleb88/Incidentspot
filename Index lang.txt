// تحديد اللغة الافتراضية
let currentLang = 'ar';

function enableUI() {
  document.getElementById('pageLoader').style.display = 'none';
  validateForm();
  
  // تحميل اللغة المحفوظة إن وجدت
  const savedLang = localStorage.getItem('preferredLanguage');
  if (savedLang) {
    currentLang = savedLang;
    toggleLanguage();
  }
}
window.addEventListener('load', enableUI);

// تبديل اللغة
function toggleLanguage() {
  currentLang = currentLang === 'ar' ? 'en' : 'ar';
  localStorage.setItem('preferredLanguage', currentLang);
  
  // تحديث النصوص
  document.getElementById('welcomeText').textContent = translations[currentLang].welcome;
  document.getElementById('subtitleText').textContent = translations[currentLang].subtitle;
  document.getElementById('idLabel').textContent = translations[currentLang].idLabel;
  document.getElementById('id').placeholder = translations[currentLang].idPlaceholder;
  document.getElementById('passwordLabel').textContent = translations[currentLang].passwordLabel;
  document.getElementById('password').placeholder = translations[currentLang].passwordPlaceholder;
  document.getElementById('loginText').textContent = translations[currentLang].login;
  document.getElementById('loadingText').textContent = translations[currentLang].loading;
  document.getElementById('copyrightText').textContent = translations[currentLang].copyright;
  document.getElementById('languageText').textContent = translations[currentLang].language;
  
  // تغيير اتجاه الصفحة بناءً على اللغة
  if (currentLang === 'ar') {
    document.dir = 'rtl';
    document.body.style.direction = 'rtl';
    document.body.style.textAlign = 'right';
    document.documentElement.lang = 'ar';
  } else {
    document.dir = 'ltr';
    document.body.style.direction = 'ltr';
    document.body.style.textAlign = 'left';
    document.documentElement.lang = 'en';
  }
}

// إضافة مستمع الحدث لزر تبديل اللغة
document.getElementById('languageSwitch').addEventListener('click', toggleLanguage);

function validateInput(input) {
  if (input.id === 'id') input.value = input.value.replace(/[^0-9]/g, '');
  validateForm();
}

function validateForm() {
  const id = document.getElementById('id').value.trim();
  const pw = document.getElementById('password').value.trim();
  document.getElementById('loginButton').disabled = !(id && pw);
}

function handleEnterKey(e) {
  if (e.key === 'Enter') { e.preventDefault(); submitForm(); }
}

function submitForm() {
  const id = document.getElementById('id').value.trim();
  const pw = document.getElementById('password').value.trim();
  const btn = document.getElementById('loginButton');

  if (!id || !pw) {
    Swal.fire({
      icon: 'warning',
      title: translations[currentLang].alertTitle,
      text: translations[currentLang].alertText,
      confirmButtonText: translations[currentLang].confirmButton
    });
    return;
  }

  btn.disabled = true;
  btn.classList.add('loading');

  fetch(GAS_ENDPOINT, {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded;charset=UTF-8' },
    body: `id=${encodeURIComponent(id)}&password=${encodeURIComponent(pw)}`
  })
  .then(res => {
    if (!res.ok) throw new Error('Network error');
    return res.json();
  })
  .then(data => {
    if (!data.ok) {
      const errorMap = {
        not_found: translations[currentLang].errorTextNotFound,
        wrong_password: translations[currentLang].errorTextWrongPassword,
        missing_fields: translations[currentLang].errorTextMissingFields,
        server_error: translations[currentLang].errorTextServer
      };
      Swal.fire({
        icon: 'error',
        title: translations[currentLang].errorTitle,
        text: errorMap[data.error] || translations[currentLang].errorTextLoginFailed,
        confirmButtonText: translations[currentLang].confirmButton
      });
    } else {
      Swal.fire({
        icon: 'success',
        title: translations[currentLang].successTitle,
        text: translations[currentLang].successText,
        showConfirmButton: false,
        timer: 1000,
        timerProgressBar: true
      }).then(() => {
        localStorage.setItem("userData", JSON.stringify(data));
        window.location.href = "main.html";
      });
    }
  })
  .catch(() => {
    Swal.fire({
      icon: 'error',
      title: translations[currentLang].connectionErrorTitle,
      text: translations[currentLang].connectionErrorText,
      confirmButtonText: translations[currentLang].confirmButton
    });
  })
  .finally(() => {
    btn.classList.remove('loading');
    btn.disabled = false;
  });
}

document.getElementById('password').addEventListener('input', validateForm);

if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('service-worker.js')
    .then(() => console.log('Service Worker Registered'));
}
